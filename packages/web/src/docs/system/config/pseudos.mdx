# Pseudos

```ts
import { pseudosSystem } from "@poodle/system";
```

```ts
const theme: SystemThemeConfig = {
	// ....
};

const listSystem = compose(pseudosSystem, ...);

export const system = createSystem(listSystem);

const styles = system({
	theme,
	_hover: {
		mt: "10px",
		color: "primary"
	},
});

// Styles result
//  { "&:hover": {marginTop: "10px", color: "cyan"} },
```
