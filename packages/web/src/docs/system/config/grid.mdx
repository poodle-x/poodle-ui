# Grid

```ts
import { girdSystem } from "@poodle/system";
```

`grid` system support CSS Grid.

```ts

const listSystem = compose(flexSystem);

export const system = createSystem(listSystem);

const styles = system({
	theme: {
		spacing: {
			scale: "1.5rem",
		},
	},
	gridTemplateRows: "50px 20px",
	gridRowGap: "scale-2"
});

// Styles result
//  { gridTemplateRows: "50px 20px" },
//  { gridRowGap: "3rem" },
```
